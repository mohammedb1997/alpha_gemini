# 🏆 تحدي الحروف - لعبة المسابقات الذكية

لعبة تفاعلية بنظام الخلايا السداسية، تجمع بين المعرفة وسرعة البديهة. تدعم نمطين للعب: **محلي** (جهاز واحد) و **مشترك** (متعدد الشاشات).

## 📌 محتويات المشروع

- `index.html` - الصفحة الرئيسية لاختيار نمط اللعب
- `normal.html` - نمط اللعب المحلي (لاعبان على نفس الجهاز)
- `judge.html` - شاشة الحكم في النمط المشترك
- `display.html` - شاشة العرض (البروجكتر) في النمط المشترك
- `player.html` - شاشة البازر للاعبين (على الهواتف)
- `style.css` - التنسيقات المشتركة
- `qbank.js` - بنك الأسئلة (40 سؤال لكل حرف)
- `howto.html` - دليل الاستخدام التفصيلي
- `README.md` - هذا الملف

---

## 🎮 كيفية التشغيل

### 1️⃣ النمط المحلي (جهاز واحد)
1. افتح ملف `index.html` واختر **"اللعب المحلي"**.
2. ستظهر لوحة الخلايا السداسية مع لوحة النتائج.
3. اضغط على أي حرف لفتح نافذة السؤال.
4. بعد قراءة السؤال:
   - استخدم **العداد التنازلي 20 ثانية**.
   - اضغط **"إظهار الإجابة"** لمعرفة الحل.
   - اضغط **"تغيير السؤال"** لسؤال آخر بنفس الحرف.
   - حدد الفائز بالضغط على **"فوز الأزرق"** أو **"فوز الأحمر"**.
5. يفوز الفريق الذي يربط بين جانبي اللوحة أولاً (أعلى-أسفل للأزرق، يسار-يمين للأحمر).

### 2️⃣ النمط المشترك (متعدد الشاشات)
1. افتح `index.html` واختر **"اللعب المشترك"** لفتح شاشة الحكم (`judge.html`).
2. في شاشة الحكم:
   - أدخل أسماء الفريقين.
   - اضغط **"إنشاء الغرفة وبدء اللعبة"**.
   - سيظهر لك **رابط شاشة البروجكتر** و **QR كود للاعبين**.
3. افتح الرابط الأول في نافذة جديدة (شاشة العرض/البروجكتر).
4. اطلب من اللاعبين مسح QR كود بهواتفهم، واختيار الفريق (أ أو ب).
5. الآن:
   - **الحكم** يختار خلية من اللوحة ليظهر السؤال.
   - **اللاعبون** يضغطون البازر في هواتفهم للإجابة.
   - **شاشة البروجكتر** تعرض السؤال وحالة البازر.
   - الحكم يقيّم الإجابة ويمنح النقطة للفريق الصحيح.

    document.getElementById('display-link').innerHTML = `<a href="${displayUrl}" target="_blank" style="color:var(--gold);">🖥️ افتح شاشة البروجكتر</a>`;
    generateQR();
  });

  peer.on('connection', (conn) => {
    connections.push(conn);
    conn.on('open', () => broadcastState());
    conn.on('data', (data) => {
      if(data.type === 'buzz') {
        document.getElementById('buzzer-status').innerText = `⚡ ${data.name} (${data.team === 'a' ? state.teamA : state.teamB})`;
        broadcast({ type: 'buzzer_won', winner: data.name, team: data.team });
      }
    });
  });
}

function initBoard() {
  const letters = Object.keys(QBANK).sort(() => 0.5 - Math.random()).slice(0, 25);
  state.board = letters.map(l => ({ letter: l, owner: null }));
  renderBoard();
}

function renderBoard() {
  const bd = document.getElementById('hex-board');
  bd.innerHTML = '';
  state.board.forEach((cell, i) => {
    const h = document.createElement('div');
    h.className = 'hex';
    if(cell.owner === 'a') h.classList.add('owned-a');
    if(cell.owner === 'b') h.classList.add('owned-b');
    if(state.activeCell === i) h.classList.add('active-q');
    h.innerText = cell.letter;
    h.onclick = () => onHexClick(i);
    bd.appendChild(h);
  });
}

function onHexClick(i) {
  if (state.board[i].owner) return;
  state.activeCell = i;
  document.getElementById('buzzer-status').innerText = 'في الانتظار...';
  
  const letter = state.board[i].letter;
  const qData = QBANK[letter][Math.floor(Math.random() * QBANK[letter].length)];
  currentQuestion = qData; // حفظ السؤال الحالي
  
  document.getElementById('q-box').innerText = qData.q;
  document.getElementById('ans-reveal').style.display = 'block';
  document.getElementById('ans-reveal').innerText = "✅ الإجابة: " + qData.a;
  
  renderBoard();
  broadcastState();
  broadcast({ type: 'show_question', q: qData.q });
}

function reopenBuzzer() {
  // إعادة فتح البازر لنفس السؤال (لا يغير السؤال)
  document.getElementById('buzzer-status').innerText = 'في الانتظار...';
  broadcast({ type: 'reset_buzzer' });
}

function judgeCorrect(team) {
  if (state.activeCell === null) return;
  state.board[state.activeCell].owner = team;
  state.scores[team]++;
  state.activeCell = null;
  currentQuestion = null;
  resetJudgeUI();
  renderBoard();
  broadcastState();
  broadcast({ type: 'reset_buzzer' });
}

function judgeSkip() {
  state.activeCell = null;
  currentQuestion = null;
  resetJudgeUI();
  renderBoard();
  broadcastState();
  broadcast({ type: 'reset_buzzer' });
}

function resetJudgeUI() {
  document.getElementById('q-box').innerText = "اختر خلية...";
  document.getElementById('ans-reveal').style.display = 'none';
  document.getElementById('buzzer-status').innerText = 'في الانتظار...';
}

function broadcast(msg) { connections.forEach(c => { if(c.open) c.send(msg); }); }
function broadcastState() { broadcast({ type: 'state_update', state: state }); }

function generateQR() {
  const playerUrl = `${window.location.origin}${window.location.pathname.replace('judge.html','player.html')}?code=${state.sessionCode}`;
  document.getElementById('qr-large').innerHTML = '';
  new QRCode(document.getElementById('qr-large'), { text: playerUrl, width: 200, height: 200 });
  document.getElementById('player-link-text').href = playerUrl;
}

        function getHexNeighbors(idx) {

                }
            }
